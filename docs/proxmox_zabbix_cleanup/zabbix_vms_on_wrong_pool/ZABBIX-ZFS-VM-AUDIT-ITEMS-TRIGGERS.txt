# Zabbix 7.0 Monitoring Configuration — Proxmox ZFS VM Placement Audit
#
# Architecture:
#   One master item reads /tmp/zfs-vm-audit.json via vfs.file.contents.
#   All other items are Dependent Items using JSONPath preprocessing.
#   No UserParameter, no system.run[], no agent scripts required.
#
# The audit script runs every 10 minutes via systemd timer.
# Zabbix polls every 5 minutes — any given poll reads the most recent output.
#
# Placement logic (blacklist / failsafe model):
#   A disk is ACCEPTABLE by default on any pool.
#   A disk is FLAGGED if its pool matches an excluded pattern.
#   The system/boot pool is always excluded by the script regardless of config.
#
# Excluded pool patterns are glob-based and case-insensitive:
#   pbs*    → pbs, PBS, pbs-fast, PBS-BACKUPS
#   rpool   → rpool exactly
#   Backup* → Backups, backup-pool, BACKUP2
#
# Severity integer mapping:
#   0 = OK       — all disks on acceptable pools
#   1 = FLAGGED  — one or more disks on an excluded pool
#  -1 = UNKNOWN  — pvesh config query failed for that guest
#
# Stale file alert: if generated_unix has not advanced in 2 hours,
# the timer or script has failed and Zabbix alerts on that independently.
#
# Host macro: {$ZFS_AUDIT_STALE_SECONDS}  default 7200 (2 hours)
# Override per host if a different stale window is needed.
#
# ============================================================================
# MASTER ITEM
# ============================================================================

Name: PVE - ZFS VM Placement Audit Report (master)
Type: Zabbix agent
Key: vfs.file.contents[/tmp/zfs-vm-audit.json]
Type of information: Text
Update interval: 5m
History: 7d
Description: >
  Reads the full JSON report written by proxmox-zfs-vm-audit.py.
  All other ZFS audit items depend on this master item.
  Script runs every 10 minutes via systemd timer; Zabbix polls every 5 minutes.

---

# ============================================================================
# TRIGGERS
# ============================================================================

Name: PVE - ZFS placement: VMs on excluded pool on {HOST.NAME}
Expression: last(/pve-host/pve.zfs.flagged.count) > 0
Severity: Warning
Manual close: Yes
Description: >
  One or more VMs or LXC containers on {HOST.NAME} have at least one disk
  residing on an excluded ZFS pool (system/boot pool or a pool matching the
  configured exclusion patterns such as pbs*, backup*, rpool).
  These disks risk filling the system pool or a backup pool.
  Review pve.zfs.flagged.detail on this host for affected VMIDs, names, and
  which disk is on which pool.
  Migrate the affected disk(s) to a dedicated VM storage pool.
  DO NOT delete data without confirming with a senior admin.

---

Name: PVE - ZFS audit: UNKNOWN guests on {HOST.NAME}
Expression: last(/pve-host/pve.zfs.unknown.count) > 0
Severity: Average
Manual close: Yes
Description: >
  One or more guests on {HOST.NAME} could not be queried by pvesh.
  This may indicate pveproxy is down, a guest is in an inconsistent state,
  or the script does not have sufficient permissions.
  Check: journalctl -u proxmox-zfs-vm-audit.service -n 50

---

Name: PVE - ZFS audit report is stale on {HOST.NAME}
Expression: (now() - last(/pve-host/pve.zfs.generated.unix)) > {$ZFS_AUDIT_STALE_SECONDS}
Severity: High
Manual close: No
Description: >
  The ZFS VM placement audit report on {HOST.NAME} has not been updated within
  {$ZFS_AUDIT_STALE_SECONDS} seconds (default 7200 = 2 hours).
  This means proxmox-zfs-vm-audit.timer has likely failed, or the script
  crashed without writing output.
  Check: systemctl status proxmox-zfs-vm-audit.timer
  Check: journalctl -u proxmox-zfs-vm-audit.service -n 30

---

Name: PVE - ZFS audit report file missing on {HOST.NAME}
Expression: last(/pve-host/vfs.file.exists[/tmp/zfs-vm-audit.json]) = 0
Severity: High
Manual close: No
Description: >
  /tmp/zfs-vm-audit.json does not exist on {HOST.NAME}.
  Normal immediately after a reboot — /tmp is volatile and the file is
  recreated when the timer fires (5 minutes post-boot).
  If this persists beyond 15 minutes after boot, investigate:
  Check: systemctl status proxmox-zfs-vm-audit.timer
  Check: journalctl -u proxmox-zfs-vm-audit.service -n 30

---

Name: PVE - ZFS audit script missing on {HOST.NAME}
Expression: last(/pve-host/vfs.file.exists[/usr/local/bin/proxmox-zfs-vm-audit.py]) = 0
Severity: High
Manual close: No
Description: >
  /usr/local/bin/proxmox-zfs-vm-audit.py has been removed from {HOST.NAME}.
  No ZFS placement auditing is occurring on this node.

---

Name: PVE - ZFS audit total guests is zero on {HOST.NAME}
Expression: >
  last(/pve-host/pve.zfs.total.vms.checked) = 0
  and last(/pve-host/pve.zfs.total.lxc.checked) = 0
  and (now() - last(/pve-host/pve.zfs.generated.unix)) < {$ZFS_AUDIT_STALE_SECONDS}
Severity: Warning
Manual close: Yes
Description: >
  The audit script ran (report is recent) but found zero VMs and zero LXC
  containers on {HOST.NAME}. On a node with running guests this indicates a
  pvesh permissions problem or a script error.
  Check: journalctl -u proxmox-zfs-vm-audit.service -n 50

---

# ============================================================================
# HOST MACRO
# ============================================================================

Macro: {$ZFS_AUDIT_STALE_SECONDS}
Default value: 7200
Description: >
  Maximum acceptable age of /tmp/zfs-vm-audit.json in seconds before the
  staleness trigger fires. Default is 2 hours (7200 seconds).
  Override at the host level for nodes with different maintenance windows.
